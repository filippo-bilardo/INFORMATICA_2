Esempi base di funzioni JavaScript

Questa pagina mostra esempi pratici di utilizzo delle funzioni JavaScript con il DOM.

1. Dichiarazione e invocazione di funzioni

function saluta(nome) {
    document.getElementById('output1').innerHTML = 
        'Ciao, ' + nome + '!';
}

function salutaPersonalizzato() {
    const nome = prompt('Come ti chiami?', 'Visitatore');
    saluta(nome);
}

2. Parametri e valore di ritorno

+ 
function somma(a, b) {
    return Number(a) + Number(b); // Convert to number
}

function calcolaEVisualizza() {
    const n1 = document.getElementById('num1').value;
    const n2 = document.getElementById('num2').value;
    const risultato = somma(n1, n2);
    
    document.getElementById('output2').innerHTML = 
        n1 + ' + ' + n2 + ' = ' + risultato;
}

3. Funzioni come valori

// Oggetto con funzioni come valori
const operazioni = {
    somma: (a, b) => a + b,
    sottrazione: (a, b) => a - b,
    moltiplicazione: (a, b) => a * b
};

function applicaOperazione(tipo) {
    const a = 10;
    const b = 5;
    
    // Seleziona la funzione dall'oggetto in base al tipo
    const operazione = operazioni[tipo];
    const risultato = operazione(a, b);
    
    document.getElementById('output3').innerHTML = 
        `Risultato dell'operazione ${tipo}: ${risultato}`;
}

4. Funzioni anonime e eventi

Conteggio: 0
// Eseguito quando il documento Ã¨ pronto
document.addEventListener('DOMContentLoaded', function() {
    let contatore = 0;
    
    document.getElementById('bottoneClick').addEventListener('click', function() {
        contatore++;
        document.getElementById('output4').innerHTML = 'Conteggio: ' + contatore;
    });
    
    document.getElementById('bottoneReset').addEventListener('click', function() {
        contatore = 0;
        document.getElementById('output4').innerHTML = 'Conteggio: ' + contatore;
    });
});

5. Arrow functions

document.addEventListener('DOMContentLoaded', () => {
    // Arrow function per trasformare il testo
    const trasformaTesto = (trasformazione) => {
        const testo = document.getElementById('inputTesto').value;
        
        let risultato;
        if (trasformazione === 'upper') {
            risultato = testo.toUpperCase();
        } else if (trasformazione === 'lower') {
            risultato = testo.toLowerCase();
        }
        
        document.getElementById('output5').innerHTML = risultato;
    };
    
    document.getElementById('bottoneUppercase').addEventListener('click', 
        () => trasformaTesto('upper'));
    
    document.getElementById('bottoneLowercase').addEventListener('click', 
        () => trasformaTesto('lower'));
});
