Animazioni Base con Canvas

Canvas è uno strumento eccellente per creare animazioni. In questa guida vedremo come creare animazioni base utilizzando la funzione requestAnimationFrame, che è il metodo più efficiente per creare animazioni fluide nel browser.

1. Animazione con requestAnimationFrame

requestAnimationFrame è un metodo che chiede al browser di eseguire una funzione di animazione prima del successivo ridisegno. Questo metodo è ottimizzato per le animazioni e sincronizzato con il refresh rate del monitor.

FPS: 0
Velocità: 
// Variabili per l'animazione
let animationId;
let x = 0;
let speed = 5;

// Funzione di animazione
function animate() {
    // Pulisci il canvas
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Aggiorna la posizione
    x += speed;
    
    // Se l'oggetto arriva al bordo, inverti la direzione
    if (x > canvas.width - 50 || x < 0) {
        speed = -speed;
    }
    
    // Disegna l'oggetto
    ctx.fillStyle = 'blue';
    ctx.fillRect(x, 100, 50, 50);
    
    // Richiedi il frame successivo
    animationId = requestAnimationFrame(animate);
}

// Avvia l'animazione
animate();

2. Animazione Controllata dal Tempo

Per creare animazioni che si comportino in modo coerente indipendentemente dalla frequenza di aggiornamento del monitor, è importante basare le animazioni sul tempo trascorso tra i frame.

FPS: 0
Velocità: 200 px/s 
// Variabili per l'animazione
let animationId;
let x = 0;
let speed = 200; // pixel al secondo
let lastTime = 0;

// Funzione di animazione
function animate(timestamp) {
    // Calcola il tempo trascorso dall'ultimo frame in secondi
    const deltaTime = (timestamp - lastTime) / 1000;
    lastTime = timestamp;
    
    // Pulisci il canvas
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Aggiorna la posizione in base al tempo
    x += speed * deltaTime;
    
    // Se l'oggetto arriva al bordo, inverti la direzione
    if (x > canvas.width - 50 || x < 0) {
        speed = -speed;
    }
    
    // Disegna l'oggetto
    ctx.fillStyle = 'green';
    ctx.fillRect(x, 100, 50, 50);
    
    // Richiedi il frame successivo
    animationId = requestAnimationFrame(animate);
}

// Avvia l'animazione
lastTime = performance.now();
animate(lastTime);

3. Animazione con Fisica

Aggiungendo semplici concetti di fisica come gravità, attrito e rimbalzo, le animazioni possono diventare molto più realistiche.

FPS: 0
Gravità: 980 px/s²
Rimbalzo: 0.7 
// Definizione di una pallina
class Ball {
    constructor(x, y, radius, color) {
        this.x = x;
        this.y = y;
        this.radius = radius;
        this.color = color;
        this.velocityX = Math.random() * 200 - 100;
        this.velocityY = Math.random() * 200 - 100;
    }
    
    // Aggiorna la posizione della pallina
    update(deltaTime, gravity, bounce) {
        // Applica la gravità
        this.velocityY += gravity * deltaTime;
        
        // Aggiorna la posizione
        this.x += this.velocityX * deltaTime;
        this.y += this.velocityY * deltaTime;
        
        // Gestisce le collisioni con i bordi
        if (this.x > canvas.width - this.radius) {
            this.x = canvas.width - this.radius;
            this.velocityX *= -bounce;
        } else if (this.x < this.radius) {
            this.x = this.radius;
            this.velocityX *= -bounce;
        }
        
        if (this.y > canvas.height - this.radius) {
            this.y = canvas.height - this.radius;
            this.velocityY *= -bounce;
        } else if (this.y < this.radius) {
            this.y = this.radius;
            this.velocityY *= -bounce;
        }
    }
    
    // Disegna la pallina
    draw(ctx) {
        ctx.fillStyle = this.color;
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
        ctx.fill();
    }
}

4. Animazione di una Scena Complessa

Con Canvas possiamo creare scene complesse combinando diverse animazioni e gestendo autonomamente ciascun elemento della scena.

FPS: 0
Numero di stelle: 200 
// Gestione di una scena con centinaia di oggetti
class SceneManager {
    constructor() {
        this.stars = [];
        this.planets = [];
        this.comets = [];
    }
    
    initialize(starCount) {
        // Crea stelle
        this.stars = [];
        for (let i = 0; i < starCount; i++) {
            this.stars.push({
                x: Math.random() * canvas.width,
                y: Math.random() * canvas.height,
                radius: Math.random() * 2 + 1,
                brightness: Math.random() * 0.8 + 0.2,
                twinkleSpeed: Math.random() * 5 + 1
            });
        }
        
        // Crea pianeti
        this.planets = [
            {
                x: canvas.width / 2,
                y: canvas.height / 2,
                radius: 30,
                color: '#3498db',
                angle: 0,
                speed: 0.3
            },
            {
                x: 0,
                y: 0,
                radius: 15,
                color: '#e74c3c',
                angle: Math.PI / 2,
                orbitRadius: 120,
                speed: 0.7
            }
        ];
        
        // Inizializza comete
        this.comets = [];
    }
    
    update(deltaTime) {
        // Aggiorna stelle (effetto brillio)
        for (let star of this.stars) {
            star.brightness = 0.2 + Math.abs(Math.sin(deltaTime * star.twinkleSpeed)) * 0.8;
        }
        
        // Aggiorna pianeti
        for (let planet of this.planets) {
            if (planet.orbitRadius) {
                planet.angle += planet.speed * deltaTime;
                planet.x = this.planets[0].x + Math.cos(planet.angle) * planet.orbitRadius;
                planet.y = this.planets[0].y + Math.sin(planet.angle) * planet.orbitRadius;
            }
        }
        
        // Aggiorna comete
        for (let i = this.comets.length - 1; i >= 0; i--) {
            const comet = this.comets[i];
            comet.x += comet.speedX * deltaTime;
            comet.y += comet.speedY * deltaTime;
            comet.lifetime -= deltaTime;
            
            if (comet.lifetime <= 0 || comet.x < -50 || comet.x > canvas.width + 50 ||
                comet.y < -50 || comet.y > canvas.height + 50) {
                this.comets.splice(i, 1);
            }
        }
        
        // Occasionalmente aggiungi una nuova cometa
        if (Math.random() < 0.02) {
            this.addComet();
        }
    }
    
    addComet() {
        const side = Math.floor(Math.random() * 4);
        let x, y, speedX, speedY;
        
        switch (side) {
            case 0: // Top
                x = Math.random() * canvas.width;
                y = -20;
                speedX = Math.random() * 100 - 50;
                speedY = Math.random() * 100 + 50;
                break;
            case 1: // Right
                x = canvas.width + 20;
                y = Math.random() * canvas.height;
                speedX = -(Math.random() * 100 + 50);
                speedY = Math.random() * 100 - 50;
                break;
            case 2: // Bottom
                x = Math.random() * canvas.width;
                y = canvas.height + 20;
                speedX = Math.random() * 100 - 50;
                speedY = -(Math.random() * 100 + 50);
                break;
            case 3: // Left
                x = -20;
                y = Math.random() * canvas.height;
                speedX = Math.random() * 100 + 50;
                speedY = Math.random() * 100 - 50;
                break;
        }
        
        this.comets.push({
            x: x,
            y: y,
            speedX: speedX,
            speedY: speedY,
            radius: Math.random() * 3 + 2,
            tailLength: Math.random() * 50 + 20,
            color: '#f39c12',
            lifetime: Math.random() * 10 + 5
        });
    }
    
    draw(ctx) {
        // Pulisci il canvas
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // Sfondo
        ctx.fillStyle = '#0c0c1a';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        
        // Disegna stelle
        for (let star of this.stars) {
            ctx.fillStyle = `rgba(255, 255, 255, ${star.brightness})`;
            ctx.beginPath();
            ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2);
            ctx.fill();
        }
        
        // Disegna comete
        for (let comet of this.comets) {
            // Disegna la coda
            const gradient = ctx.createLinearGradient(
                comet.x, comet.y,
                comet.x - comet.speedX * (comet.tailLength / 100),
                comet.y - comet.speedY * (comet.tailLength / 100)
            );
            gradient.addColorStop(0, `rgba(255, 255, 255, 0.8)`);
            gradient.addColorStop(1, 'rgba(255, 255, 255, 0)');
            
            ctx.strokeStyle = gradient;
            ctx.lineWidth = comet.radius;
            ctx.beginPath();
            ctx.moveTo(comet.x, comet.y);
            ctx.lineTo(
                comet.x - comet.speedX * (comet.tailLength / 100),
                comet.y - comet.speedY * (comet.tailLength / 100)
            );
            ctx.stroke();
            
            // Disegna la testa
            ctx.fillStyle = comet.color;
            ctx.beginPath();
            ctx.arc(comet.x, comet.y, comet.radius, 0, Math.PI * 2);
            ctx.fill();
        }
        
        // Disegna pianeti
        for (let planet of this.planets) {
            ctx.fillStyle = planet.color;
            ctx.beginPath();
            ctx.arc(planet.x, planet.y, planet.radius, 0, Math.PI * 2);
            ctx.fill();
            
            // Se è un pianeta in orbita, mostra l'orbita
            if (planet.orbitRadius) {
                ctx.strokeStyle = 'rgba(255, 255, 255, 0.2)';
                ctx.beginPath();
                ctx.arc(this.planets[0].x, this.planets[0].y, planet.orbitRadius, 0, Math.PI * 2);
                ctx.stroke();
            }
        }
    }
}

5. Animazione con easing

Le funzioni di easing rendono le animazioni più naturali e piacevoli: 


                // 5. Animazione con easing
                const easingCanvas = document.getElementById('easingCanvas');
                const easingCtx = easingCanvas.getContext('2d');
                let easingAnimationId;
            
                const easingFunctions = {
                    linear: t => t,
                    easeOut: t => 1 - Math.pow(1 - t, 3),
                    easeInOut: t => t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2,
                    bounce: t => {
                        const n1 = 7.5625;
                        const d1 = 2.75;
                        if (t < 1 / d1) return n1 * t * t;
                        else if (t < 2 / d1) return n1 * (t -= 1.5 / d1) * t + 0.75;
                        else if (t < 2.5 / d1) return n1 * (t -= 2.25 / d1) * t + 0.9375;
                        else return n1 * (t -= 2.625 / d1) * t + 0.984375;
                    }
                };
            
                function animateWithEasing(easingFunction) {
                    if (easingAnimationId) cancelAnimationFrame(easingAnimationId);
                    
                    let startTime = null;
                    const duration = 2000;
                    const startX = 50;
                    const endX = easingCanvas.width - 50;
            
                    function step(timestamp) {
                        if (!startTime) startTime = timestamp;
                        const elapsed = timestamp - startTime;
                        const progress = Math.min(elapsed / duration, 1);
                        const easedProgress = easingFunction(progress);
                        
                        easingCtx.clearRect(0, 0, easingCanvas.width, easingCanvas.height);
                        easingCtx.beginPath();
                        easingCtx.arc(
                            startX + (endX - startX) * easedProgress,
                            easingCanvas.height / 2,
                            20, 0, Math.PI * 2
                        );
                        easingCtx.fillStyle = 'purple';
                        easingCtx.fill();
            
                        if (progress < 1) {
                            easingAnimationId = requestAnimationFrame(step);
                        }
                    }
                    easingAnimationId = requestAnimationFrame(step);
                }
            
                document.getElementById('linearEasing').addEventListener('click', () => animateWithEasing(easingFunctions.linear));
                document.getElementById('easeOutEasing').addEventListener('click', () => animateWithEasing(easingFunctions.easeOut));
                document.getElementById('easeInOutEasing').addEventListener('click', () => animateWithEasing(easingFunctions.easeInOut));
                document.getElementById('bounceEasing').addEventListener('click', () => animateWithEasing(easingFunctions.bounce));
            
                // Disegna pallina iniziale
                easingCtx.beginPath();
                easingCtx.arc(50, easingCanvas.height / 2, 20, 0, Math.PI * 2);
                easingCtx.fillStyle = 'purple';
                easingCtx.fill();

Suggerimenti per Animazioni Performanti

  • Usa requestAnimationFrame: è sincronizzato con il refresh rate del monitor ed è più efficiente rispetto a setInterval o setTimeout.
  • Fai animazioni basate sul tempo: assicura che la velocità dell'animazione sia coerente su dispositivi diversi.
  • Limita i calcoli: evita operazioni costose in ogni frame.
  • Ottimizza i disegni: disegna solo ciò che è cambiato.
  • Gestisci le risorse: pulisci gli oggetti non più necessari.
